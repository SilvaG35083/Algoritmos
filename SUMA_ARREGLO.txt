// Algoritmo para sumar los elementos de un arreglo A de tamanio n
ALGORITMO_SUMA(A[n])
begin
    // Declaracion de variables locales
    total <- 0;
    
    // El ciclo FOR se ejecuta n veces
    for i <- 1 to n do 
    begin
        // Operacion O(1) dentro del ciclo
        total <- total + A[i]; 
    end;
    
    // (Fin del algoritmo)
end

// Algoritmo MergeSort (Ordenamiento por Mezcla)
// Estructura tipica de Divide y Venceras
// T(n) = 2T(n/2) + O(n) -> Aplica Teorema Maestro
MERGE_SORT(A, n)
begin
    If (n > 1) then
    begin
        mitad <- n / 2;
        
        // 1. Dividir: Dos llamadas recursivas
        // Usamos CALL explÃ­cito para que el parser no falle
        CALL MERGE_SORT(A, mitad);
        CALL MERGE_SORT(A, mitad);
        
        // 2. Conquistar/Combinar (Coste Lineal O(n))
        for i <- 1 to n do
        begin
            temp <- A[i];
        end;
    end;
end

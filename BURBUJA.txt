// Algoritmo de Ordenamiento Burbuja
// Nota: Usa n como el tamaño
BURBUJA(A, n)
begin
    // Ciclo externo
    for i <- 1 to n-1 do
    begin
        // Ciclo interno
        // El límite (n-i) depende del ciclo externo
        for j <- 1 to n-i do
        begin
            // Bloque de costo O(1)
            If (A[j] > A[j+1]) then
            begin
                temp <- A[j];
                A[j] <- A[j+1];
                A[j+1] <- temp;
            end;
        end;
    end;
end
